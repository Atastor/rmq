def add_named(key, font_name_or_font, size = nil)
  font = if font_name_or_font.is_a?(UIFont)
    font_name_or_font
  else
    Font.font_with_name(font_name_or_font, size || 22)
  end

  Font.define_singleton_method(key) do
    font
  end
end
