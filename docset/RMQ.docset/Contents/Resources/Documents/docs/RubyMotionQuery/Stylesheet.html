def initialize(controller)
  @controller = RubyMotionQuery::RMQ.weak_ref(controller)

  unless Stylesheet.application_was_setup
    Stylesheet.application_was_setup = true
    application_setup 
  end
  setup
end

    @controller.view.contentSize
  else
    CGSizeZero
  end
